GATE(XOR, XOR2)
GATE(AND, AND1)
GATE(AND, AND2)
GATE(OR, OR1)

INPUT(A, AND2, 0)
INPUT(B, XOR1, 1)
INPUT(B, AND2, 1)
INPUT(C_IN, XOR2, 1)
INPUT(C_IN, AND1, 1)
CONN(XOR1, 0, XOR2, 0)
CONN(XOR1, 0, AND1, 0)
CONN(AND1, 0, OR1, 0)
CONN(AND2, 0, OR1, 1)
OUTPUT(XOR2, 0, SUM)
OUTPUT(OR1, 0, C_OUT)
